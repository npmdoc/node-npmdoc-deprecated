function deprecated.field (msg, log, parent, field, val)

  • description and source-code
    field = function (msg, log, parent, field, val) {
  var called = false;
  var getter = function(){
    if (!called) {
      called = true;
      log(msg);
    }
    return val;
  };
  var setter = function(v) {
    if (!called) {
      called = true;
      log(msg);
    }
    val = v;
    return v;
  };
  Object.defineProperty(parent, field, {
    get: getter,
    set: setter,
    enumerable: true
  });
  return;
}
